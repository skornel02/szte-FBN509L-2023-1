U_b_calc = @(I_f, T, C_p) (I_f .* T) ./ C_p;
 
freq = 50;
U_f = 9;
R_t = 1000;
T = 1 / freq;
C_p = 22 * 10^-6;
I_f = U_f / R_t
I_f = 0.0090
 
U_b = U_b_calc(I_f, T, C_p)
